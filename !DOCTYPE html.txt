       function startApp(song) {

    // 🎧 LOAD SPOTIFY IN SAME PAGE
    const frame = document.getElementById('spotifyFrame');
    const box = document.getElementById('spotifyEmbed');

    frame.src = songLinks[song];
    box.classList.remove('hidden');

    // (optional) stop unused audio tag
    const audio = document.getElementById('myAudio');
    if (audio) audio.pause();

    document.getElementById('s0').classList.add('hidden');
    document.getElementById('mainContent').classList.remove('hidden');

    confetti({ particleCount: 150, spread: 70, origin: { y: 0.6 } });
}

        function nextSection(id) {
            document.querySelectorAll('section').forEach(s => s.classList.add('hidden'));
            document.getElementById(id).classList.remove('hidden');
            window.scrollTo({ top: 0, behavior: 'smooth' });
            if(id === 's2') {
                setTimeout(() => {
                    document.getElementById('meterBar').style.width = '100%';
                    setTimeout(() => document.getElementById('meterResult').classList.remove('hidden'), 3100);
                }, 500);
            }
        }

        let heartTaps = 0;
        function repairHeart() {
            heartTaps++;
            const heart = document.getElementById('brokenHeart');
            if(heartTaps < 5) heart.style.transform = `scale(${1 + heartTaps * 0.1})`;
            if(heartTaps === 5) {
                heart.innerText = '❤️';
                heart.classList.add('glow-text', 'scale-125');
                document.getElementById('tapStatus').innerText = 'તેં મારું દિલ જોડી દીધું! 🥰';
                document.getElementById('puzBtn').classList.remove('hidden');
                confetti({ particleCount: 100, shapes: ['heart'] });
            }
        }

        let sorryCount = 0;
        function flipSorry() {
            const inner = document.getElementById('sorryInner');
            const txt1 = document.getElementById('sorryTxt1');
            const txt2 = document.getElementById('sorryTxt2');
            
            inner.classList.toggle('flipped');
            sorryCount++; // 1 TAP = 1 COUNT
            document.getElementById('sorryCounter').innerText = `${sorryCount} / 5000`;
            
            const randomLine = sorryLines[Math.floor(Math.random() * sorryLines.length)];
            if(inner.classList.contains('flipped')) {
                txt2.innerText = randomLine;
            } else {
                txt1.innerText = randomLine;
            }
            
            if(sorryCount % 5 === 0) confetti({ particleCount: 15, gravity: 2, scalar: 0.7 });
        }

        function handleResponse(type) {
            const msgBox = document.getElementById('finalMessage');
            const text = document.getElementById('messageText');
            msgBox.classList.remove('hidden');
            if(type === 'yes') {
                text.innerText = "આઈ લવ યુ સો મચ! તેં મને માફ કરીને દુનિયાનો સૌથી સુખી વ્યક્તિ બનાવી દીધો! ❤️✨";
            } else {
                text.innerText = "હવે આપણે ક્યારેય અલગ નહીં થઈએ! તું જ મારું સર્વસ્વ છે સુહવિન! 💞🫂";
            }
            confetti({ particleCount: 500, spread: 160, scalar: 2 });
        }
